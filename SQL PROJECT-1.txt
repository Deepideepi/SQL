--DQL COMMENTS
--1)List all students who enrolled in the course  MYSQL Basics .
SELECT S.FULL_NAME,C.COURSE_NAME
FROM ENROLLMENTS E,STUDENTS S,COURSES C
WHERE E.STUDENT_ID=S.STUDENT_ID
AND E.COURSE_ID=C.COURSE_ID
AND C.COURSE_NAME='MYSQL';

--2)Show the total number of students enrolled in each course.
SELECT C.COURSE_NAME,COUNT(C.COURSE_NAME) AS COURSE_COUNT
FROM ENROLLMENTS E,COURSES C,STUDENTS S
WHERE S.STUDENT_ID=E.STUDENT_ID
AND C.COURSE_ID=E.COURSE_ID
GROUP BY C.COURSE_NAME;

--3)Display all courses with their instructor names.
SELECT C.COURSE_NAME,I.FULL_NAME
FROM COURSES C,INSTRUCTORS I
WHERE C.INSTRUCTOR_ID=I.INSTRUCTOR_ID;

--4)Find students who joined after 27-JULY-2025.
SELECT FULL_NAME,JOIN_DATE
FROM STUDENTS
WHERE JOIN_DATE>TO_DATE('2025-07-27','YYYY-MM-DD');

--5)List instructors who teach more than one course.
SELECT I.FULL_NAME
FROM INSTRUCTORS I,COURSES S
WHERE I.INSTRUCTOR_ID=I.INSTRUCTOR_ID
GROUP BY(I.FULL_NAME)
HAVING COUNT( I.FULL_NAME)>1;


--6)Show all ACTIVE enrollments.
SELECT S.FULL_NAME, E.STATUS
FROM ENROLLMENTS E,STUDENTS S
WHERE E.STATUS='ACTIVE' AND E.STUDENT_ID=S.STUDENT_ID;

--7)Show courses with duration greater than 2 months.
SELECT COURSE_NAME,DURATION_MONTHS
FROM COURSES
WHERE DURATION_MONTHS>'2 MONTHS';

--8)Display student names with the courses they enrolled in.
SELECT S.FULL_NAME,C.COURSE_NAME
FROM ENROLLMENTS E,COURSES C,STUDENTS S
WHERE S.STUDENT_ID=E.STUDENT_ID
AND C.COURSE_ID=E.COURSE_ID;

--9)Find the highest instructor salary.
SELECT FULL_NAME,SALARY
FROM INSTRUCTORS
WHERE
SALARY=(SELECT MAX(SALARY)FROM INSTRUCTORS);

--10)Count enrollments grouped by status (ACTIVE / COMPLETED / CANCELLED)
SELECT STATUS, COUNT(STATUS) AS STATUS_COUNT
FROM ENROLLMENTS
WHERE
   STATUS IN ('ACTIVE', 'COMPLETED', 'CANCELLED')
GROUP BY
    STATUS;
